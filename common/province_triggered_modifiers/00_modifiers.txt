# Province Triggered modifiers are here.
# These are added to provinces through the add_province_triggered_modifier effect
#
# Effects are fully scriptable here.

# Scope: province(ROOT)
# This trigger fires once each month, use it to handle stuff that 
# depend on monthly updates (can use country and global scope)
#
monthly_pulse = {
	potential = {
		always = yes
	}

	trigger = {
		custom_trigger_tooltip = {
			tooltip = MONTHLY_PULSE_TT 
			always = yes
		}
	}
	
	local_production_efficiency = 0.15
	local_tax_modifier = 0.3 
	
	# Gets called on each new day
	on_activation = {
		if = { 
			limit = { NOT = { check_variable = { which = month_pulse_eta value = 1 } } }
			owner = { log = "EVE: Monthly pulse for [This.GetName] just happened." }
			set_variable = { which = month_pulse_eta value = 30 }
		
			# Do this when we run the mod for the FIRST TIME this campaign.
			# It has to be run from here because the ROOT scope is set to capital province.
			if = {
				limit = { NOT = { had_global_flag = { flag = eve_initialized days = 32 } } }
				owner = { recalculate_province_min_autonomy = yes }
			}
			# This triggers when the capital province has changed (for any reason)
			if = {
				limit = { NOT = { is_capital = yes } }
				owner = { 
					recalculate_province_min_autonomy = yes
					capital_scope = {
						log = "EVE: [Prev.GetName] has changed its capital to [This.GetName]."  #Debug
						add_province_triggered_modifier = monthly_pulse
					}
				}
			}
		}
		else = { 
			subtract_variable = { which = month_pulse_eta value = 1 }
			#log = "EVE: [Prev.GetName]: month pulse eta: [Root.month_pulse_eta.GetValue]."  #Debug
		}
	}
	
	# This should only fire when we remove the modifier via "remove_province_modifier"
	on_deactivation = {
		owner = { 
			log = "EVE: Monthly pulse deactivated."
		}
	}
	
}
