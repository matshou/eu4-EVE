#Advisor report on estimated time to increase rating
#Ex: "estimates it will take us..."
defined_text = {
	name = policy_gov_increase_time_txt
	random = no

	text = {
		localisation_key = string_null		# No Advisor
		trigger = {
			is_var_null = { var = policy_gov_adv_m }
		}
	}
	text = {
		localisation_key = string_policy_gov_increase_time_1
		trigger = {
			OR = {
				governance.is_rating_max = no
				governance.becoming_neutral = yes
			}
		}
	}
	text = {
		localisation_key = string_policy_gov_increase_time_2	# Reached max
		trigger = {
			always = yes
		}
	}
}
